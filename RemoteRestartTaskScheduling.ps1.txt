# Invoke this script to set Windows Machine restart remotely from your RDP/jump server
$server = "<your server>"

# Desired task start date and time - this works for both 12h and 24h formats
$RebootDate = Get-Date "2025-11-01 09:00PM"

# Start of actual function
$scriptBlock = {

	# This line is crucial - if you add spaces in comment or use double quotation instead of apotrophe and quotation will brick your task
    $A = New-ScheduledTaskAction -Execute "shutdown.exe" -Argument "/r /f /t 0 /c '<place_comment_in_apostrophes_and_without_spaces_please>'"
	
	# Using "using" variable because RebootDate variable is outside of this script block so value is not known inside this script block
    $T = New-ScheduledTaskTrigger -Once -At $using:RebootDate
	
	# Set schedule to be ran as "SYSTEM", run whenever user is logged on or not and running task with highest privileges
    $P = New-ScheduledTaskPrincipal -UserId "SYSTEM" -LogonType ServiceAccount -RunLevel Highest
	# Putting parameters in place
    $D = New-ScheduledTask -Action $A -Principal $P -Trigger $T

	# Create actual task - put name manually below
    Register-ScheduledTask "Example Task" -InputObject $D
}

# Execute commands from script block on machine pointed in first lines
Invoke-Command -ComputerName $server -ScriptBlock $scriptBlock
